const DEFAULT_RUNGS = [1080, 720, 480, 360];

export function sanitizeName(input) {
  const file = String(input).replace(/\\/g, '/').split('/').pop();
  const base = file.replace(/\.[^.]+$/, '');
  const slug = base.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '');
  return slug || 'video';
}

export function deriveLadder(sourceHeight, override) {
  const even = (h) => h - (h % 2);
  const src = even(sourceHeight);
  let rungs;
  if (override && override.length) {
    rungs = [...new Set(override.map(even))].filter((h) => h > 0 && h <= src);
  } else {
    rungs = DEFAULT_RUNGS.filter((h) => h <= src);
    if (rungs.length === 0) rungs = [src];
  }
  return rungs.sort((a, b) => b - a);
}
```

const AVC_KBPS = { 2160: 14000, 1440: 9000, 1080: 5000, 720: 2800, 480: 1400, 360: 800 };
const CODEC_STRINGS = { avc: 'avc1.640028', av1: 'av01.0.08M.08', hevc: 'hvc1.1.6.L120.90' };

export function bitrateFor(height, codec) {
  let base = AVC_KBPS[height];
  if (!base) {
    const keys = Object.keys(AVC_KBPS).map(Number).sort((a, b) => a - b);
    const k = keys.find((x) => x >= height) ?? keys[keys.length - 1];
    base = AVC_KBPS[k];
  }
  const factor = codec === 'av1' || codec === 'hevc' ? 0.6 : 1;
  const kbps = Math.round(base * factor);
  return { kbps, maxrate: Math.round(kbps * 1.07), bufsize: kbps * 2 };
}

export function codecString(codec) {
  const s = CODEC_STRINGS[codec];
  if (!s) throw new Error(`Unbekannter Codec: ${codec}`);
  return s;
}

export function parseProbe(json) {
  const streams = json.streams || [];
  const v = streams.find((s) => s.codec_type === 'video');
  if (!v) throw new Error('Keine Video-Spur in der Quelle gefunden.');
  const hasAudio = streams.some((s) => s.codec_type === 'audio');
  const rate = v.avg_frame_rate && v.avg_frame_rate !== '0/0' ? v.avg_frame_rate : v.r_frame_rate || '25/1';
  const [num, den] = rate.split('/').map(Number);
  const fps = den ? num / den : 25;
  const duration = Number(v.duration || json.format?.duration || 0);
  const bitrate = Number(json.format?.bit_rate || v.bit_rate || 0);
  return {
    width: Number(v.width),
    height: Number(v.height),
    fps: Math.round(fps) || 25,
    duration,
    hasAudio,
    bitrate,
  };
}

const BOOL_FLAGS = new Set(['gpu', 'force']);
const VALID_CODECS = new Set(['avc', 'av1', 'hevc']);

export function parseArgs(argv) {
  const positional = [];
  const flags = {};
  for (let i = 0; i < argv.length; i++) {
    const a = argv[i];
    if (a.startsWith('--')) {
      const key = a.slice(2);
      if (BOOL_FLAGS.has(key)) { flags[key] = true; continue; }
      flags[key] = argv[++i];
    } else {
      positional.push(a);
    }
  }
  const [input, outdir] = positional;
  if (!input || !outdir) throw new Error('Aufruf: hls.mjs <input> <outdir> [flags]');

  const mode = flags.mode || 'background';
  if (!['background', 'content'].includes(mode)) throw new Error(`Ungültiger --mode: ${mode}`);

  const codecs = (flags.codecs || 'avc,av1').split(',').map((s) => s.trim()).filter(Boolean);
  for (const c of codecs) if (!VALID_CODECS.has(c)) throw new Error(`Unbekannter Codec: ${c}`);
  if (!codecs.includes('avc')) codecs.unshift('avc');

  const ladder = flags.ladder
    ? flags.ladder.split(',').map((s) => parseInt(s, 10)).filter((n) => n > 0)
    : null;

  return {
    input,
    outdir,
    mode,
    codecs,
    ladder,
    gpu: !!flags.gpu,
    seg: parseInt(flags.seg || '4', 10),
    poster: parseFloat(flags.poster || '0'),
    name: flags.name || null,
    force: !!flags.force,
  };
}
```

  if (codec === 'avc') return gpu ? ['-c:v', 'h264_nvenc', '-preset', 'p5'] : ['-c:v', 'libx264', '-preset', 'medium'];

    ? ['-c:v', 'hevc_nvenc', '-preset', 'p5', '-tag:v', 'hvc1']
    : ['-c:v', 'libx265', '-preset', 'medium', '-tag:v', 'hvc1'];
  return ['-c:v', 'libsvtav1', '-preset', '6']; // av1: immer CPU
}

export function ffmpegArgsFor({ input, codec, height, fps, seg, gpu, dir }) {
  const d = String(dir).replace(/\\/g, '/');
  const even = (h) => h - (h % 2);
  const g = Math.max(1, Math.round(fps * seg));
  const { kbps, maxrate, bufsize } = bitrateFor(height, codec);
  return [
    '-y', '-i', input,
    '-an',
    '-vf', `scale=-2:${even(height)}`,
    ...videoEncoder(codec, gpu),
    '-b:v', `${kbps}k`, '-maxrate', `${maxrate}k`, '-bufsize', `${bufsize}k`,
    '-g', String(g), '-keyint_min', String(g), '-sc_threshold', '0',
    '-force_key_frames', `expr:gte(t,n_forced*${seg})`,
    '-hls_time', String(seg), '-hls_playlist_type', 'vod',
    '-hls_segment_type', 'fmp4', '-hls_fmp4_init_filename', 'init.mp4',
    '-hls_segment_filename', `${d}/seg-%05d.m4s`,
    `${d}/${height}p.m3u8`,
  ];
}

export function audioArgsFor({ input, seg, dir }) {
  const d = String(dir).replace(/\\/g, '/');
  return [
    '-y', '-i', input,
    '-vn', '-c:a', 'aac', '-b:a', '128k', '-ac', '2',
    '-hls_time', String(seg), '-hls_playlist_type', 'vod',
    '-hls_segment_type', 'fmp4', '-hls_fmp4_init_filename', 'init.mp4',
    '-hls_segment_filename', `${d}/seg-%05d.m4s`,
    `${d}/audio.m3u8`,
  ];
}

export function buildMasterPlaylist(variants, opts = {}) {
  const audio = opts.audio || null;
  const lines = ['#EXTM3U', '#EXT-X-VERSION:7'];
  if (audio) {
    lines.push(
      `#EXT-X-MEDIA:TYPE=AUDIO,GROUP-ID="aud",NAME="${audio.name || 'audio'}",DEFAULT=YES,AUTOSELECT=YES,URI="${audio.uri}"`
    );
  }
  for (const v of variants) {
    const codecs = audio ? `${codecString(v.codec)},mp4a.40.2` : codecString(v.codec);
    const bw = v.kbps * 1000 + (audio ? 128000 : 0);
    const attrs = [
      `BANDWIDTH=${bw}`,
      `AVERAGE-BANDWIDTH=${bw}`,
      `RESOLUTION=${v.width}x${v.height}`,
      `CODECS="${codecs}"`,
    ];
    if (audio) attrs.push('AUDIO="aud"');
    lines.push(`#EXT-X-STREAM-INF:${attrs.join(',')}`);
    lines.push(v.uri);
  }
  return lines.join('\n') + '\n';
}

export function embedHtml({ name, mode }) {
  const attrs = mode === 'background' ? 'muted loop autoplay playsinline' : 'controls playsinline';
  return `<!doctype html>
<html lang="de">
<head><meta charset="utf-8"><title>${name}</title>
<style>html,body{margin:0;background:#000}video{width:100%;height:auto;display:block}</style></head>
<body>
<!-- Hinweis: hls.js wird hier per CDN geladen. Für Produktion hls.js selbst hosten
     oder auf eine feste Version mit integrity/SRI pinnen. -->
<video id="v" poster="poster.jpg" ${attrs}></video>
<script src="https://cdn.jsdelivr.net/npm/hls.js@1"></script>
<script>
  var v = document.getElementById('v'), src = 'master.m3u8';
  if (v.canPlayType('application/vnd.apple.mpegurl')) { v.src = src; }
  else if (window.Hls && Hls.isSupported()) { var h = new Hls(); h.loadSource(src); h.attachMedia(v); }
</script>

```js
#!/usr/bin/env node
import { spawnSync } from 'node:child_process';
import { existsSync, mkdirSync, writeFileSync, rmSync } from 'node:fs';
import { join } from 'node:path';
import * as L from './lib.mjs';

function hasBinary(bin) {
  const r = spawnSync(bin, ['-version'], { encoding: 'utf8' });
  return r.status === 0;
}

function run(bin, args) {
  const r = spawnSync(bin, args, { stdio: 'inherit' });
  if (r.status !== 0) throw new Error(`${bin} fehlgeschlagen (Exit ${r.status})`);
}

function probe(input) {
  const r = spawnSync('ffprobe', ['-v', 'quiet', '-print_format', 'json', '-show_format', '-show_streams', input], { encoding: 'utf8' });
  if (r.status !== 0) throw new Error('ffprobe fehlgeschlagen: ' + input);
  return L.parseProbe(JSON.parse(r.stdout));
}

function fail(msg) {
  console.error('Fehler: ' + msg);
  process.exit(1);
}

function main() {
  let cfg;
  try {
    cfg = L.parseArgs(process.argv.slice(2));
  } catch (e) {
    fail(e.message);
  }
  if (!hasBinary('ffmpeg') || !hasBinary('ffprobe')) fail('ffmpeg/ffprobe nicht im PATH gefunden.');
  if (!existsSync(cfg.input)) fail('Input nicht gefunden: ' + cfg.input);

  const meta = probe(cfg.input);
  let mode = cfg.mode;
  if (mode === 'content' && !meta.hasAudio) {
    console.warn('Hinweis: Quelle hat keine Audiospur → background-Modus.');
    mode = 'background';
  }

  const name = L.sanitizeName(cfg.name || cfg.input);
  const root = join(cfg.outdir, name);
  if (existsSync(root)) {
    if (!cfg.force) fail(`Zielordner existiert bereits: ${root} (mit --force überschreiben).`);
    rmSync(root, { recursive: true, force: true });
  }
  mkdirSync(root, { recursive: true });

  const ladder = L.deriveLadder(meta.height, cfg.ladder);
  const widthFor = (h) => {
    const w = Math.round((h * meta.width) / meta.height);
    return w - (w % 2);
  };

  const variants = [];
  for (const codec of cfg.codecs) {
    for (const h of ladder) {
      const dir = join(root, codec, `${h}p`);
      mkdirSync(dir, { recursive: true });
      console.log(`Encode ${codec} ${h}p …`);
      run('ffmpeg', L.ffmpegArgsFor({ input: cfg.input, codec, height: h, fps: meta.fps, seg: cfg.seg, gpu: cfg.gpu, dir }));
      variants.push({ codec, width: widthFor(h), height: h, kbps: L.bitrateFor(h, codec).kbps, uri: `${codec}/${h}p/${h}p.m3u8` });
    }
  }

  let audioOpt = null;
  if (mode === 'content') {
    const adir = join(root, 'audio');
    mkdirSync(adir, { recursive: true });
    console.log('Encode Audio (AAC) …');
    run('ffmpeg', L.audioArgsFor({ input: cfg.input, seg: cfg.seg, dir: adir }));
    audioOpt = { uri: 'audio/audio.m3u8', name: 'audio' };
  }

  writeFileSync(join(root, 'master.m3u8'), L.buildMasterPlaylist(variants, { audio: audioOpt }));
  run('ffmpeg', ['-y', '-ss', String(cfg.poster), '-i', cfg.input, '-frames:v', '1', '-q:v', '3', join(root, 'poster.jpg')]);
  writeFileSync(join(root, 'embed.html'), L.embedHtml({ name, mode }));

  console.log(`Fertig: ${root}`);
}

main();
```

# video-hls — Videos für Web-Auslieferung sequenzieren (HLS / EXTM3U)

Wandelt **ein** Quellvideo (`.mp4`/`.webm`) in adaptives **HLS** um: ABR-Leiter,
fMP4/CMAF-Segmente, `#EXTM3U`-Master-Playlist, Poster und Embed-Snippet — alles in
einem eigenen Ordner pro Video.

## Voraussetzungen

- `ffmpeg` und `ffprobe` 7.x im PATH (mit `libx264`, `libsvtav1`, optional `libx265`/NVENC).
- Node.js ≥ 20.

## Aufruf

```bash
node "${CLAUDE_PLUGIN_ROOT}/skills/video-hls/scripts/hls.mjs" <input> <outdir> [flags]
```

Oder bequem über den Command `/video-hls <input> <outdir> [flags]`.

Es entsteht `<outdir>/<videoname>/` mit der gesamten HLS-Struktur.

## Flags

| Flag        | Default               | Bedeutung |
|-------------|-----------------------|-----------|
| `--mode`    | `background`          | `background` = kein Audio, loop/autoplay-optimiert · `content` = mit AAC-Audio + Controls |
| `--codecs`  | `avc,av1`             | Komma-Liste aus `avc`, `av1`, `hevc`. `avc` wird immer als Fallback erzwungen. |
| `--gpu`     | aus                   | Nutzt NVENC für `avc`/`hevc` (schnell). `av1` bleibt CPU (`libsvtav1`). |
| `--ladder`  | aus Quelle abgeleitet | Override der Höhen, z. B. `1080,720,480`. Kein Upscaling. |
| `--seg`     | `4`                   | Segmentlänge in Sekunden. |
| `--poster`  | `0`                   | Zeitstempel (s) für `poster.jpg`. |
| `--name`    | aus Dateiname         | Override des Ordnernamens (wird sanitisiert). |
| `--force`   | aus                   | Überschreibt einen bereits existierenden Zielordner. |

## Ablauf

1. `ffprobe` analysiert Auflösung, FPS, Dauer, Audio-Spur.
2. ABR-Leiter aus der Quellhöhe ableiten (Standard `1080/720/480/360`, nie hochskalieren).
3. Ordner `<outdir>/<name>/` anlegen.
4. Je Codec × Rung ein keyframe-aligned `ffmpeg`-Lauf → CMAF/fMP4 (sequentiell, RAM-schonend).
5. `content`-Modus: zusätzlich eine AAC-Audio-Rendition.
6. `master.m3u8` (`#EXTM3U`, Multi-Codec/-Auflösung) schreiben.
7. `poster.jpg` + `embed.html` erzeugen.
